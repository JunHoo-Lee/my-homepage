% Research Interests
\section{Research Interests}
{\normalsize
I am interested in foundation models, especially in the scaling behavior of
learned representations: what internal structure emerges as pretraining scales,
what remains stable across model and data scale, and how these structures shape
downstream behavior, adaptation, and controllability. My work connects model
design, pretraining, representation learning, evaluation, and inference-time
control across language, vision, generative, and multimodal settings. More
broadly, I aim to understand not only how models become stronger with scale,
but also what structure makes those gains possible and reliable.
}

    {CVPR 2026}
    {\#Diffusion Model \#Fingerprinting}
    {\myname, \coauthors{Mijin Koo, Nojun Kwak}}
    {Developed a query-only black-box attribution framework for fine-tuned text-to-image models, combining compositional semantic probing with Bayesian lineage inference to identify model family and origin across diverse fine-tuned variants.}

    {What's Making That Sound Right Now? Video-centric Audio-Visual Localization}
    {ICCV 2025}
    {\#Audio-Visual Localization}
    {\coauthors{Hahyeon Choi}, \myname, \coauthors{Nojun Kwak}}
    {Presented AVATAR, a video-centric audio-visual localization benchmark, together with TAVLO, a temporally aware model designed for realistic grounding scenarios involving mixed sounds, multiple entities, off-screen sources, and cross-event mismatches.}

\selpub
    {Deep Support Vectors}
    {NeurIPS 2024}
    {\#Deep Learning Theory}
    {\myname, \coauthors{Hyunho Lee, Kyomin Hwang, Nojun Kwak}}
    {Introduced DeepKKT, a method for recovering support-like learned structures from pretrained classifiers, enabling explanation, practical dataset distillation, model inversion, and latent-space generative editing without access to the original training data.}

\selpub
    {Do Not Think About Pink Elephant! (Co-first)}
    {CVPR Workshops 2024}
    {\#Diffusion Model \#Safety}
    {\coauthors{Kyomin Hwang$^\dagger$, Suyoung Kim$^\dagger$}, \myname$^\dagger$, \coauthors{Nojun Kwak}}
    {Studied how large generative models fail to represent negation, revealing a ``white bear phenomenon'' in which suppressed concepts remain active, and proposed prompt-based attack and defense strategies grounded in representation-space analysis.}

\selpub

    {\#Meta Learning}
